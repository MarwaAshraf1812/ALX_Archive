int check_line(char *line)
{
    int is_usable = -1;
    if (line == NULL)
        return (is_usable);
    char *line_dup = strdup(line);
    char *word = strtok(line_dup, " \n");
    if (word[0] != "#")
        is_usable = 0;
    free(line_dup);
    free(word);
    return (is_usable);
}
